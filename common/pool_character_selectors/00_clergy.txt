pool_court_chaplain = {
	valid_character = {
		# Capable adults
		is_capable_adult = yes
		is_imprisoned = no
		is_married = no
		is_betrothed = no
		# In minimum good standing with faith
		NOT = { has_trait = excommunicated }

		# Correct gender
		scope:base.faith = { has_allowed_gender_for_clergy = root }
		
		# No close family members for temporal religions unless they have relevant traits
		trigger_if = {
			limit = {
				scope:base.faith = {
					NOT = { has_doctrine_parameter = lay_clergy }
				}
			}
			OR = {
				NOT = { is_close_family_of = scope:base }
				has_trait = devoted
				has_trait = saint
			}
		}
	}
	
	character_score = {
		base = 1
		
		modifier = {
			add = 100
			has_same_culture_as = scope:base
		}
		modifier = {
			add = 10
			has_same_culture_group_as = scope:base
		}
		modifier = {
			factor = 1.5
			age > 40
		}
		modifier = {
			factor = 3
			scope:base.faith = { has_preferred_gender_for_clergy = root }
		}
	}
	
	config = {
		background = clergy
		age = { 25 55 }
	}
	
	selection_count = 25
	
	pool = faith
	gender = clergy
}

pool_theocratic_succession = {
	valid_character = {
		is_capable_adult = yes
		scope:base.faith = { has_allowed_gender_for_clergy = root }
		is_imprisoned = no
		is_married = no
		is_betrothed = no
		has_children = no
		NOT = { has_trait = excommunicated }
	}
	
	character_score = {
		base = 1
		
		modifier = {
			add = 1000
			has_same_culture_as = scope:base
		}
		modifier = {
			add = 500
			has_same_culture_group_as = scope:base
		}
		modifier = {
			dynasty.dynasty_prestige_level >= 2
			add = {
				value = dynasty.dynasty_prestige_level
				multiply = 20
				subtract = 20
			}
		}
		modifier = {
		piety_level >= 2
			add = {
				value = piety_level
				multiply = 5
			}
		}
		modifier = {
			factor = 1.5
			age > 40
		}
		modifier = {
			factor = 1.5
			age > 60
		}
		modifier = {
			factor = 3
			scope:base.faith = { has_preferred_gender_for_clergy = root }
		}
	}
	
	config = {
		background = clergy
		age = { 30 70 }
	}
	
	selection_count = 25
	
	pool = faith
	gender = clergy
}
