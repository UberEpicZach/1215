armor_value_radiantplate = 75
 

armor_value = {
    value = 0
    if = {limit = {has_trait = warform} add = armor_value_warform}
    if = {limit = {has_character_modifier = artifact_shardplate_modifier} add = armor_value_shardplate}
    else_if = {limit = {has_character_modifier = summoned_radiantplate_modifier} add = armor_value_radiantplate}
    else_if = {limit = {has_character_modifier = artifact_28_modifier} add = armor_value_28}
    else_if = {limit = {has_character_modifier = artifact_29_modifier} add = armor_value_29}
    else_if = {limit = {has_character_modifier = artifact_30_modifier} add = armor_value_30}
    else_if = {limit = {has_character_modifier = artifact_31_modifier} add = armor_value_31}
}

armor_increase_mult = {value = armor_value multiply = 0.01 add = 1}
armor_half_increase_mult = {value = armor_value multiply = 0.005 add = 1}
armor_decrease_mult = {value = 100 subtract = armor_value multiply = 0.01}
armor_half_decrease_mult = {value = 200 subtract = armor_value multiply = 0.005}
