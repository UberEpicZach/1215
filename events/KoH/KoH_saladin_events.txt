		every_ruler = {

			inheritable = no

				type = trebuchet
				stacks = 10
			}

			location = capital_province
		}

		spawn_army = {
			uses_supply = no
			inheritable = no
			name = saladins_fanatics
			men_at_arms = {
				type = saladin_riders
				stacks = 25
			}
			levies = 1
			location = capital_province
		}

		spawn_army = {
			uses_supply = no
			inheritable = no
			name = saladins_fanatics
			men_at_arms = {
				type = saladin_riders
				stacks = 25
			}
			levies = 1
			location = capital_province
		}

		spawn_army = {
			uses_supply = no
			inheritable = no
			name = saladins_fanatics
			men_at_arms = {
				type = saladin_riders
				stacks = 10
			}
			levies = 1
			location = capital_province
		}
		
		scope:jerusalem_ruler = {
			trigger_event = KoH_saladin.2
		}
		
		every_character_war = {
            limit = {
                primary_defender = title:k_jerusalem.holder
                primary_attacker = title:e_arabia.holder
            }

		add_defender = global_var:hospitaller_holy_order_title.holder
		}
	}
}
